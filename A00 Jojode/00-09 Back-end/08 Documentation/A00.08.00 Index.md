# `$= '../' + dv.current().file.folder.split('/').at(-1) + '<br /><br />'`


```dataviewjs

const currentPage = dv.current();

const files = app.vault.getFiles().filter(
	file => file.path.includes(dv.current().file.folder) && 
	file.path.split('.').at(-1) !== 'md' &&
	/^([A-Z][0-9][0-9].[0-9][0-9].[0-9][0-9])/.test(file.path.split('/').at(-1))
).map(file => ({ link: dv.fileLink(file.path), path: file.path }));

const pages = dv.pages()
	.where(page => page.file.folder === currentPage.file.folder)
	.sort(page => page.file.name, 'ASC');

dv.span([...files, ...pages].sort(compare).map(item => {
	if (item.file)
		item = item.file.name === currentPage.file.name ? '<b>' + item.file.name + '</b>' : item.file.link;
	else item = item.link;
	
	return '<span>' + item + ' </span>';
}));

function compare(a, b) {
  a = a.file ? { path: a.file.path } : a;
  b = b.file ? { path: b.file.path } : b;
  a = a.path.split('/').at(-1).split('.')[2];
  b = b.path.split('/').at(-1).split('.')[2];
  
  if (a < b) return -1;
  if (a > b) return 1;
  return 0;
}
```
